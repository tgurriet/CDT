398 template <typename T, typename TVertexIter, typename TGetVertexCoord>
399 DuplicatesInfo FindDuplicates(
400  TVertexIter first,
401  TVertexIter last,
402  TGetVertexCoord getX,
403  TGetVertexCoord getY)
404 {
405  typedef unordered_map<V2d<T>, std::size_t> PosToIndex;
406  PosToIndex uniqueVerts;
407  const std::size_t verticesSize = std::distance(first, last);
408  DuplicatesInfo di = {
409  std::vector<std::size_t>(verticesSize), std::vector<std::size_t>()};
410  for(std::size_t iIn = 0, iOut = iIn; iIn < verticesSize; ++iIn, ++first)
411  {
412  typename PosToIndex::const_iterator it;
413  bool isUnique;
414  tie(it, isUnique) = uniqueVerts.insert(
415  std::make_pair(V2d<T>::make(getX(*first), getY(*first)), iOut));
416  if(isUnique)
417  {
418  di.mapping[iIn] = iOut++;
419  continue;
420  }
421  di.mapping[iIn] = it->second; // found a duplicate
422  di.duplicates.push_back(iIn);
423  }
424  return di;
425 }
